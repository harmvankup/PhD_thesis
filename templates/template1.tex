% -----------------------------------------------
% CUSTOM TITLE PAGE (Mimicking "reedthesis" style)
% -----------------------------------------------
$if(title)$
  \thispagestyle{empty}
  \begin{titlepage}
    \centering
    \vspace*{2cm}
    {\Large \textbf{\@title}\par}
    \vspace{1cm}
    {\large \@division \\
    \@institution \\
    for the degree of \\
    \@degree \\}
    \vspace{2cm}
    {\large \@author \\
    \@department\par}
    \vspace{1.5cm}
    Advisor: \@advisor \\
    $if(altadvisor)$
      Co-Advisor: \@altadvisor
    $endif$
    \vfill
    {\large \@date}
  \end{titlepage}
  \cleardoublepage
$endif$

% If a "quote" was provided, show it on a separate page
$if(startquote)$

  \thispagestyle{empty}
  \begin{flushright}

  \end{flushright}
